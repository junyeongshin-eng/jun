const STORAGE_KEY_TOKEN = 'salesmap_api_token'; const STORAGE_KEY_CURSOR_PREFIX = 'salesmap_cursor_deal_history_'; document.addEventListener('DOMContentLoaded', () => { const savedToken = localStorage.getItem(STORAGE_KEY_TOKEN); if (savedToken) { document.getElementById('apiToken').value = savedToken; } document.getElementById('dealId').addEventListener('change', loadCursor); document.getElementById('historyForm').addEventListener('submit', onSubmit); loadCursor(); }); function storageKeyForDeal(dealId) { return `${STORAGE_KEY_CURSOR_PREFIX}${dealId || 'unknown'}`; } function loadCursor() { const dealId = document.getElementById('dealId').value.trim(); const savedCursor = localStorage.getItem(storageKeyForDeal(dealId)); document.getElementById('cursor').value = savedCursor || ''; } function saveCursor(dealId, cursor) { if (!dealId) return; const key = storageKeyForDeal(dealId); if (cursor) { localStorage.setItem(key, cursor); } else { localStorage.removeItem(key); } } async function onSubmit(event) { event.preventDefault(); resetStatus(); const token = document.getElementById('apiToken').value.trim(); const dealId = document.getElementById('dealId').value.trim(); const limit = document.getElementById('limit').value.trim(); const cursor = document.getElementById('cursor').value.trim(); const startDate = document.getElementById('startDate').value; const endDate = document.getElementById('endDate').value; const baseUrl = document.getElementById('baseUrl').value.trim(); if (!token || !dealId || !baseUrl) { showError('필수 입력값(API 토큰, 딜 ID, 엔드포인트)을 모두 입력하세요.'); return; } localStorage.setItem(STORAGE_KEY_TOKEN, token); const params = new URLSearchParams({ deal_id: dealId }); if (limit) params.set('limit', limit); if (cursor) params.set('cursor', cursor); if (startDate) params.set('start_date', new Date(startDate).toISOString()); if (endDate) params.set('end_date', new Date(endDate).toISOString()); const url = `${baseUrl}?${params.toString()}`; const submitBtn = document.getElementById('submitBtn'); submitBtn.disabled = true; submitBtn.textContent = '⏳ 조회 중...'; try { setStatus('요청을 전송하는 중입니다...'); const response = await fetch(url, { headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}`, }, }); const text = await response.text(); let json; try { json = JSON.parse(text); } catch (parseError) { throw new Error('JSON 파싱에 실패했습니다. 응답 본문을 확인하세요.'); } if (!response.ok) { throw new Error(json.message || json.error || `HTTP ${response.status}`); } document.getElementById('responseViewer').textContent = JSON.stringify(json, null, 2); setStatus('✅ 조회가 완료되었습니다. 필요 시 next_cursor 를 사용해 이어서 조회하세요.'); if (json.next_cursor) { saveCursor(dealId, json.next_cursor); document.getElementById('cursor').value = json.next_cursor; } else { saveCursor(dealId, ''); } } catch (error) { showError(error.message); } finally { submitBtn.disabled = false; submitBtn.textContent = '🚀 히스토리 조회'; } } function setStatus(message) { const status = document.getElementById('status'); status.innerHTML = ` ${message} `; } function resetStatus() { document.getElementById('status').textContent = ''; hideError(); } function showError(message) { const errorBox = document.getElementById('errorBox'); errorBox.textContent = message; errorBox.style.display = 'block'; setStatus('⚠️ 오류가 발생했습니다. 아래 메시지를 확인하세요.'); } function hideError() { const errorBox = document.getElementById('errorBox'); errorBox.style.display = 'none'; errorBox.textContent = ''; }
